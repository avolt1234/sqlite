public float createStats(String score1, String score2){

        //Used for testing purposes
        //String score1 = "1.323432";
        //String score2 = "0.32123";

        //Convert the Strings into floats
        float score_1 = Float.parseFloat(score1);
        float score_2 = Float.parseFloat(score2);

        //Put the score on a 0 to -10 scale
        float finalConv = -5 - (score_1 + score_2);

        //Convert the scale to 0 to 10 and find a percentage
        float finalScore = finalConv * -10;

        //Return final percentage
        return finalScore;
    }
